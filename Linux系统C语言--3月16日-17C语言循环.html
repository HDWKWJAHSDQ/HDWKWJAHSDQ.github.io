4, #include <> 与 #include "" 的区别？
• #include <头文件.h>：用于包含系统标准头文件，编译器优先去系统指定目录（如 /usr/include）查找。
• #include "头文件.h"：用于包含用户自定义头文件，编译器优先在当前源文件所在目录查找，找不到再去系统目录查找。

4.统计并打印n以内含有m的数(n从键盘上输入，m从键盘上输入,m要小于n)
思路：
输入初始化：读取 n 和 m，初始化计数变量 cnt=0；
遍历检查：循环遍历 1~n 的每个数 i，对每个 i 用临时变量 num=i、tmp=m 逐位拆解；
逐位匹配：对比 num 和 tmp 的最后一位，匹配则 tmp 右移一位，直到 tmp 为 0（说明 m 完整匹配），标记 flag=1；不匹配则重置 tmp=m；
统计输出：flag=1 时打印 i 并 cnt++，最后输出总数 cnt。

思路1：(太麻烦了，不采用，刚开始)
#include <stdio.h>

int main()
{
int i, cnt, n, m; // i变量用于遍历n到m, cnt用于统计包含有m的数的个数，n和m变量用于键盘输入
cnt = n = m = 0;
scanf("%d", &n);
scanf("%d", &m);

for(i = 1; i <= n; i++)
{
int num = i; // 每次用临时变量num去保存遍历到的数字
int flag = 0; // 标志位，判断i是否包含m

while(num != 0) // 外层循环：控制比较的“起始位置”
{
int current_num = num; // 【关键修改】：用一个新的变量代替 num 去试错
int tmp = m; // 重新拿出完整的 m 准备比较

// 内层循环：开始具体的逐位比较
while(tmp != 0 && current_num != 0)
{
if(current_num % 10 == tmp % 10) // 如果当前位匹配
{
current_num /= 10;
tmp /= 10;
}
else
{
break; // 【关键修改】：一旦某一位不匹配，直接打断内层比较，不再继续
}
}

// 检查刚才的内层循环是因为什么结束的
if(tmp == 0) // 如果 tmp 被除到了 0，说明 m 的每一位都匹配成功了
{
flag = 1;
break; // 找到了就直接退出外层循环
}

num /= 10; // 【关键修改】：如果刚才没匹配成功，num 丢掉最后一位，下一次从新的位置开始比较
}

if(flag) // 如果最后flag = 1
{
printf("%d\n", i); // 就打印这个数
cnt++; // 统计个数++
}
}
printf("%d\n", cnt); // 把个数打印输出
return 0;
}

goto语句的作用：
goto 是一个无条件跳转语句，核心作用是：直接跳转到代码中预先标记好的位置执行，跳过中间代码或回到前面的代码段。
场景：适合多层循环退出、统一错误处理等场景。一般不往前跳，而是往后跳

#include <stdio.h>

int main() {
// 1. 定义标记：标记名 + 冒号（标记名自定义，如here、loop）
start:
printf("Hello\n");

int n = 1;
if (n == 1) {
// 2. goto + 标记名：跳转到标记位置
goto start;
}

return 0;
}
上面的代码会无限打印 “Hello”（因为n=1永远成立，一直跳回start）
